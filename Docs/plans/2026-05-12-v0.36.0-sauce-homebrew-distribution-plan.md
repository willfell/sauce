# v0.36.0 — Sauce Homebrew Distribution Implementation Plan

> **For the implementing agent:** REQUIRED SUB-SKILL: Use executing-plans to implement this plan task-by-task.

**Goal:** Move `pantry/` out of synced vaults into a Homebrew-installed location and distribute `sauce` via personal tap `willfell/homebrew-sauce`. Eliminates dirty-tree + sync-corruption failures on second machines.

**Architecture:** Brew formula installs pantry tarball into `$(brew --prefix)/Cellar/sauce/<v>/libexec/`. A `~/.sauce/vaults.json` per-machine registry tracks installed vaults. New CLI verbs (`reinstall`, `migrate-layout`, `doctor`, `link`/`unlink`, `vault add|list|remove`) layer onto existing `sauce-cli.js` dispatcher. `~/.sauce/active-pantry` symlink supports dev mode (`sauce link`). Two-verb update flow: `brew upgrade sauce && sauce reinstall --all`.

**Tech Stack:** Node 18+, zero runtime deps beyond `@inquirer/prompts` (already present). Ruby/Homebrew formula in tap repo. GitHub Actions for release automation. Tap repo: `willfell/homebrew-sauce` (already created at `/Users/willfell/Documents/obsidian/sync/homebrew-sauce`).

**Design doc:** `Docs/plans/2026-05-12-sauce-homebrew-distribution-design.md` (commit `a259870`). Read it first.

**Cycle hygiene reminders (per CLAUDE.md):**
- Conventional-commits messages, commit per task, push to `origin/main` after each commit.
- No `Co-authored-by: Claude` trailer; author is the user.
- All metadata is JSON, not YAML.
- All paths use `{{template_variables}}` in mechanism/blueprint code.
- Bump `platform/manifest.json` `workshop_version` at the end of S1 (the first stage that changes platform code).

// platform/cli/registry.js — per-machine vault registry at ~/.sauce/vaults.json.
// Atomic writes (temp + rename). Read-tolerant of missing file.

const fs = require("fs");
const path = require("path");
const os = require("os");

function registryDir() { return path.join(os.homedir(), ".sauce"); }
function registryPath() { return path.join(registryDir(), "vaults.json"); }

function read() {
    const p = registryPath();
    if (!fs.existsSync(p)) return { version: 1, vaults: [] };
    try { return JSON.parse(fs.readFileSync(p, "utf8")); }
    catch (_e) { return { version: 1, vaults: [] }; }
}

function write(reg) {
    const dir = registryDir();
    if (!fs.existsSync(dir)) fs.mkdirSync(dir, { recursive: true });
    const final = registryPath();
    const tmp = final + ".tmp";
    fs.writeFileSync(tmp, JSON.stringify(reg, null, 2) + "\n");
    fs.renameSync(tmp, final);
}

function add(vaultPath) {
    const abs = path.resolve(vaultPath);
    const reg = read();
    if (reg.vaults.some(v => v.path === abs)) return reg; // dedupe
    reg.vaults.push({ path: abs, registered_at: new Date().toISOString() });
    write(reg);
    return reg;
}

function remove(vaultPath) {
    const abs = path.resolve(vaultPath);
    const reg = read();
    reg.vaults = reg.vaults.filter(v => v.path !== abs);
    write(reg);
    return reg;
}

function list() { return read().vaults; }

function pruneMissing() {
    const reg = read();
    const kept = [], removed = [];
    for (const v of reg.vaults) {
        if (fs.existsSync(v.path)) kept.push(v); else removed.push(v.path);
    }
    reg.vaults = kept;
    write(reg);
    return removed;
}

module.exports = { read, write, add, remove, list, pruneMissing, registryPath };
```

// cmd-vault.js — sauce vault add | list | remove.
const path = require("path");
const registry = require("./registry.js");

async function run(ctx, args) {
    const sub = (args || [])[0];
    if (sub === "add") {
        const p = path.resolve((args || [])[1] || ctx.vaultPath);
        registry.add(p);
        console.log(`  Registered: ${p}`);
        return;
    }
    if (sub === "list") {
        const vaults = registry.list();
        if (vaults.length === 0) { console.log("  No vaults registered."); return; }
        for (const v of vaults) console.log(`  ${v.path}  (registered ${v.registered_at})`);
        return;
    }
    if (sub === "remove") {
        const p = path.resolve((args || [])[1]);
        if (!p) throw new Error("usage: sauce vault remove <path>");
        registry.remove(p);
        console.log(`  Removed: ${p}`);
        return;
    }
    throw new Error("usage: sauce vault <add|list|remove> [path]");
}

module.exports = { run };
```

// cmd-reinstall.js — re-run installer against --vault or --all registered vaults.
const path = require("path");
const registry = require("./registry.js");

async function run(ctx, args) {
    const flags = args || [];
    const all = flags.includes("--all");
    const vaultIdx = flags.indexOf("--vault");
    const keepMissing = flags.includes("--keep-missing");

    let targets = [];
    if (all) {
        if (!keepMissing) {
            const removed = registry.pruneMissing();
            if (removed.length) console.log(`  Pruned ${removed.length} missing vault(s): ${removed.join(", ")}`);
        }
        targets = registry.list().map(v => v.path);
    } else if (vaultIdx >= 0 && flags[vaultIdx + 1]) {
        targets = [path.resolve(flags[vaultIdx + 1])];
    } else {
        throw new Error("usage: sauce reinstall (--all | --vault <path>)");
    }

    if (targets.length === 0) { console.log("  No vaults to reinstall."); return; }

    for (const t of targets) {
        console.log(`  Reinstalling: ${t}`);
        await _runInstaller(ctx, { vaultPath: t });
    }
}

async function _runInstaller(ctx, opts) {
    if (typeof ctx._runInstaller === "function") return ctx._runInstaller(opts);
    const bootstrap = require("../bootstrap.js");
    await bootstrap.phaseRunInstaller(opts);
}

module.exports = { run };
```

// Pure-function mirror of the shim's resolution logic.
// The shim is bash but the logic is unit-tested here against the same inputs.
const fs = require("fs");
const path = require("path");

function resolve({ home, brewPrefix, statSync }) {
    statSync = statSync || fs.statSync;
    const active = path.join(home, ".sauce/active-pantry");
    try {
        const s = statSync(active);
        if (s.isDirectory()) return active;
    } catch (_e) { /* missing */ }
    return path.join(brewPrefix, "libexec");
}

module.exports = { resolve };
```

**Files (workshop repo):**
- Create: `.github/workflows/release.yml`

**Step 1:** Write:

```yaml
name: release
on:
  push:
    tags: ['v*.*.*']
jobs:
  bump-tap:
    runs-on: ubuntu-latest
    steps:
      - name: Compute tarball sha256
        id: sha
        run: |
          VERSION="${GITHUB_REF_NAME#v}"
          URL="https://github.com/willfell/sauce/archive/refs/tags/${GITHUB_REF_NAME}.tar.gz"
          SHA=$(curl -fsSL "$URL" | shasum -a 256 | awk '{print $1}')
          echo "version=$VERSION" >> "$GITHUB_OUTPUT"
          echo "sha=$SHA"         >> "$GITHUB_OUTPUT"
          echo "url=$URL"         >> "$GITHUB_OUTPUT"
      - name: Checkout tap
        uses: actions/checkout@v4
        with:
          repository: willfell/homebrew-sauce
          token: ${{ secrets.TAP_PR_TOKEN }}
          path: tap
      - name: Patch formula
        run: |
          cd tap
          sed -i "s|url \".*\"|url \"${{ steps.sha.outputs.url }}\"|" Formula/sauce.rb
          sed -i "s|sha256 \".*\"|sha256 \"${{ steps.sha.outputs.sha }}\"|" Formula/sauce.rb
      - name: Open PR
        uses: peter-evans/create-pull-request@v6
        with:
          token: ${{ secrets.TAP_PR_TOKEN }}
          path: tap
          branch: bump-v${{ steps.sha.outputs.version }}
          commit-message: "feat: sauce v${{ steps.sha.outputs.version }} formula"
          title: "sauce v${{ steps.sha.outputs.version }}"
          body: |
            Auto-generated bump from `willfell/sauce` tag push.
            URL: ${{ steps.sha.outputs.url }}
            SHA: ${{ steps.sha.outputs.sha }}
```

**Step 2:** Commit + push.
