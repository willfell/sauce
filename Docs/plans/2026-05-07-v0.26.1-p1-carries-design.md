### 3.2 Helper signature + posture

NEW `async function applyAppSettings(tp, workshopManifest, history, git)` — workshop-manifest-level (runs ONCE per install run, NOT per-item). Mirrors `applyCustomJsSettings` v0.24.0 posture.

```js
async function applyAppSettings(tp, workshopManifest, history, git) {
  if (!workshopManifest || !workshopManifest.app_settings ||
      Object.keys(workshopManifest.app_settings).length === 0) {
    return;  // silent no-op when no settings declared
  }

  let basePath;
  try { basePath = tp.app.vault.adapter.getBasePath(); }
  catch (e) {
    new Notice(`applyAppSettings: vault adapter getBasePath unavailable (${e.message})`, 8000);
    history.push({ step: "app_settings", action: "error",
      message: e.message, attempted_at: new Date().toISOString(), git_commit: git.commit });
    return;
  }

  const obsidianDir = require("path").join(basePath, ".obsidian");
  if (!require("fs").existsSync(obsidianDir)) {
    history.push({ step: "app_settings", action: "skipped_obsidian_dir_absent",
      attempted_at: new Date().toISOString(), git_commit: git.commit });
    return;
  }

  const target = require("path").join(obsidianDir, "app.json");
  let existing = {};
  if (require("fs").existsSync(target)) {
    try {
      const raw = require("fs").readFileSync(target, "utf8");
      const parsed = JSON.parse(raw);
      if (typeof parsed !== "object" || parsed === null || Array.isArray(parsed)) {
        new Notice(`applyAppSettings: ${target} parsed but not a JSON object; skipping`, 8000);
        history.push({ step: "app_settings", action: "error",
          message: "malformed shape", attempted_at: new Date().toISOString(), git_commit: git.commit });
        return;
      }
      existing = parsed;
    } catch (e) {
      new Notice(`applyAppSettings: ${target} malformed JSON (${e.message}); skipping`, 8000);
      history.push({ step: "app_settings", action: "error",
        message: e.message, attempted_at: new Date().toISOString(), git_commit: git.commit });
      return;
    }
  }

  // Backup-on-edit BEFORE write (only if file existed; create-from-scratch needs no backup)
  if (require("fs").existsSync(target)) {
    try { require("fs").copyFileSync(target, target + ".sauce-backup"); }
    catch (e) {
      new Notice(`applyAppSettings: backup write failed (${e.message}); aborting`, 8000);
      history.push({ step: "app_settings", action: "error",
        message: `backup failed: ${e.message}`, attempted_at: new Date().toISOString(), git_commit: git.commit });
      return;
    }
  }

  // Additive shallow merge: platform-as-overrider for declared keys.
  const merged = { ...existing, ...workshopManifest.app_settings };

  // Atomic write
  const tmp = target + ".tmp";
  try {
    require("fs").writeFileSync(tmp, JSON.stringify(merged, null, 2), "utf8");
    require("fs").renameSync(tmp, target);
  } catch (e) {
    new Notice(`applyAppSettings: write failed (${e.message})`, 8000);
    history.push({ step: "app_settings", action: "error",
      message: e.message, attempted_at: new Date().toISOString(), git_commit: git.commit });
    return;
  }

  history.push({ step: "app_settings", action: "applied",
    keys_written: Object.keys(workshopManifest.app_settings),
    attempted_at: new Date().toISOString(), git_commit: git.commit });
}
```

### 6.3 wizard auto-add convenience

`bootstrap-lib/wizard.js` — NEW exported helper `_autoAddConvenienceIfDvBlueprintsSelected`:

```js
function _autoAddConvenienceIfDvBlueprintsSelected(selectedMechs, selectedBlueprints, manifestEntries) {
  if (selectedMechs.includes("convenience")) return selectedMechs;
  const dvBlueprints = (selectedBlueprints || []).filter(name => {
    const bp = (manifestEntries.blueprints || []).find(b => b.name === name);
    if (!bp) return false;
    // bp may be wizard's lightweight {name, version} OR full manifest with depends_on.
    // Resolve via workshopManifest.blueprints[].depends_on (lookup by name).
    const full = (manifestEntries.fullBlueprints || []).find(b => b.name === name);
    if (!full || !Array.isArray(full.depends_on)) return false;
    return full.depends_on.some(d => d.name === "convenience");
  });
  if (dvBlueprints.length === 0) return selectedMechs;
  console.log(`[info] Auto-added convenience because ${dvBlueprints[0]} depends on it.`);
  return [...selectedMechs, "convenience"];
}
```

**Wiring detail:** workshop manifest's `blueprints[]` entries are `{name, version, path}` only. The full per-blueprint manifest (with `depends_on`) lives at `<workshop>/platform/blueprints/<name>/manifest.json`. The wizard already loads workshop manifest; design phase verifies whether it also reads per-blueprint manifests OR if a one-shot read is needed in this helper. Plan refines.

Called from BOTH:
- `bootstrap.js` interactive checkbox path (after user finishes mech-select).
- `bootstrap-lib/wizard.js` `nonInteractive` defaults branch.
