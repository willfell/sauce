Add this function definition between `applyVendoredThemes` and `applyAppearance`:

```javascript
// applySnippets — for each item that declares manifest.snippets[], copy the
// source CSS to <vault>/.obsidian/snippets/<name>.css. Mirrors applyVendoredThemes
// posture: sha256-compare overwrite-with-backup (`.sauce-backup` suffix),
// failure-loud history, never-throws. Platform-vendored snippets only —
// scoped to `sauce-*.css` filenames by manifest convention (carve-out
// codified in landmine #12 v0.41.0 amendment). NEW v0.41.0.
async function applySnippets(tp, manifest, workshopPath, targetPath, history, git) {
  if (!manifest || !Array.isArray(manifest.snippets) || manifest.snippets.length === 0) return;
  const adapter = tp.app.vault.adapter;
  const fs = require("fs");
  const fsp = fs.promises;
  const path = require("path");
  const crypto = require("crypto");

  const sha256 = (buf) => crypto.createHash("sha256").update(buf).digest("hex");

  for (const entry of manifest.snippets) {
    const valid =
      entry &&
      typeof entry.source === "string" && entry.source.length > 0 &&
      typeof entry.name === "string" && entry.name.length > 0 &&
      /^sauce-[A-Za-z0-9._-]+$/.test(entry.name);

    if (!valid) {
      new Notice(`applySnippets: ${manifest.name} invalid snippet entry; skipped`, 6000);
      if (history) {
        history.push({
          event: "warning",
          step: "snippets",
          name: manifest.name,
          message: "invalid_entry",
          git_commit: git.commit,
          git_tag: git.tag,
          git_dirty: git.dirty,
          attempted_at: new Date().toISOString(),
        });
      }
      continue;
    }

    const srcAbs = path.join(workshopPath, "platform", targetPath, entry.source);
    if (!fs.existsSync(srcAbs)) {
      new Notice(`applySnippets: source absent ${srcAbs} for ${manifest.name}/${entry.name}`, 8000);
      if (history) {
        history.push({
          event: "error",
          step: "snippets",
          name: manifest.name,
          message: `source absent: ${srcAbs}`,
          git_commit: git.commit,
          git_tag: git.tag,
          git_dirty: git.dirty,
          attempted_at: new Date().toISOString(),
        });
      }
      continue;
    }

    let srcBody;
    try {
      srcBody = await fsp.readFile(srcAbs);
    } catch (e) {
      new Notice(`applySnippets: read source failed for ${srcAbs} (${e.message})`, 8000);
      if (history) {
        history.push({
          event: "error",
          step: "snippets",
          name: manifest.name,
          message: `read source failed: ${e.message}`,
          git_commit: git.commit,
          git_tag: git.tag,
          git_dirty: git.dirty,
          attempted_at: new Date().toISOString(),
        });
      }
      continue;
    }

    // Ensure dest dir exists
    const destRel = `.obsidian/snippets/${entry.name}.css`;
    const destAbsViaAdapter = destRel;  // adapter is vault-rooted
    try {
      await adapter.mkdir(".obsidian/snippets");
    } catch (e) {
      // mkdir already-exists is fine; only fail-loud on hard errors
      if (!/exists/i.test(e.message || "")) {
        new Notice(`applySnippets: mkdir .obsidian/snippets failed (${e.message})`, 8000);
        if (history) {
          history.push({
            event: "error",
            step: "snippets",
            name: manifest.name,
            message: `mkdir failed: ${e.message}`,
            git_commit: git.commit,
            git_tag: git.tag,
            git_dirty: git.dirty,
            attempted_at: new Date().toISOString(),
          });
        }
        continue;
      }
    }

    const exists = await adapter.exists(destAbsViaAdapter);
    if (exists) {
      let existingRaw;
      try {
        existingRaw = await adapter.read(destAbsViaAdapter);
      } catch (e) {
        new Notice(`applySnippets: read dest failed for ${destRel} (${e.message})`, 8000);
        if (history) {
          history.push({
            event: "error",
            step: "snippets",
            name: manifest.name,
            message: `read dest failed: ${e.message}`,
            git_commit: git.commit,
            git_tag: git.tag,
            git_dirty: git.dirty,
            attempted_at: new Date().toISOString(),
          });
        }
        continue;
      }

      const existingHash = sha256(Buffer.from(existingRaw, "utf8"));
      const srcHash = sha256(srcBody);
      if (existingHash === srcHash) {
        if (history) {
          history.push({
            event: "info",
            step: "snippets",
            name: manifest.name,
            snippet: entry.name,
            action: "skipped_identical",
            git_commit: git.commit,
            git_tag: git.tag,
            git_dirty: git.dirty,
            attempted_at: new Date().toISOString(),
          });
        }
        continue;
      }

      // Divergent: backup before overwrite (non-empty body only)
      if (existingRaw && existingRaw.length > 0) {
        try {
          await adapter.write(`${destAbsViaAdapter}.sauce-backup`, existingRaw);
        } catch (e) {
          new Notice(`applySnippets: backup write failed for ${destRel}.sauce-backup (${e.message}); aborting overwrite`, 8000);
          if (history) {
            history.push({
              event: "error",
              step: "snippets",
              name: manifest.name,
              message: `backup write failed: ${e.message}`,
              git_commit: git.commit,
              git_tag: git.tag,
              git_dirty: git.dirty,
              attempted_at: new Date().toISOString(),
            });
          }
          continue;
        }
      }
    }

    try {
      await adapter.write(destAbsViaAdapter, srcBody.toString("utf8"));
    } catch (e) {
      new Notice(`applySnippets: dest write failed for ${destRel} (${e.message})`, 8000);
      if (history) {
        history.push({
          event: "error",
          step: "snippets",
          name: manifest.name,
          message: `dest write failed: ${e.message}`,
          git_commit: git.commit,
          git_tag: git.tag,
          git_dirty: git.dirty,
          attempted_at: new Date().toISOString(),
        });
      }
      continue;
    }

    if (history) {
      history.push({
        event: "info",
        step: "snippets",
        name: manifest.name,
        snippet: entry.name,
        action: exists ? "overwrote" : "applied",
        git_commit: git.commit,
        git_tag: git.tag,
        git_dirty: git.dirty,
        attempted_at: new Date().toISOString(),
      });
    }
  }
}
```

## What to do if a stage fails

| Failure | Recovery |
| --- | --- |
| Preflight RED at any stage | Re-run `npm run release:preflight 2>&1 \| tee /tmp/preflight.log`. Find the first failing harness; if it's `run-helper-cases.js`, look for SN1/SN2/SN3 + any cross-affected case (e.g., a sub-assert counting line that lags). Don't proceed until green. |
| `applySnippets` test fails after implementation | The helper conforms to the test — DON'T change the test. Re-read applyVendoredThemes; check edge cases (mkdir exists-already; backup suffix; sha compare on bytes vs string). |
| `sauce reinstall` mid-cycle fails | Capture the full Notice / history.log diff. Most common cause: a missing manifest field (e.g., forgot to add `snippets[]` to convenience manifest) — re-check S3.2. |
| `git push --tags` fails | NEVER use `--force` against `origin v0.41.0` unless the tag was created accidentally. If pushed-then-broken, post-fix should be a v0.41.1 PATCH. |
| Workshop self-install touches MORE than the 18 allowlist paths | STOP. Investigate which helper wrote outside the allowlist. This is a landmine #12 violation and must be fixed before tag. |
