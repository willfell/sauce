```javascript
async function walkClaudeSurface(vault, options) {
  // Inputs:
  //   vault: vault root absolute path
  //   options: { reportFile, verbose, strictExit }
  //
  // Outputs: { findings: [{severity, kind, path, message, ...}], counts: {dead_path, orphan, stale, consumer_edit_at_risk, aligned} }
  //
  // Algorithm:
  // 1. Read ranch/claude-surface-registry.json. If missing → emit one finding severity=dead_path, abort.
  // 2. For each item in registry.contributions, for each entry:
  //    - Check dest exists on filesystem. If not → severity=dead_path.
  //    - For claude_md_row entries: resolve row.path (if present), check FS.
  // 3. Walk .claude/commands/**/*.md and .claude/skills/**/SKILL.md on disk.
  //    For each file not present in registry → severity=orphan.
  // 4. For each registry entry that did exist on disk: read body, search for <!-- @claude-surface:version <X> -->. If present and X != registry.version → severity=stale_but_valid.
  // 5. For each registry entry: compare deployed file content to blueprint source. If differs AND no matching .local/ shadow exists → severity=consumer_edit_at_risk.
  // 6. Aligned count = registry entries with no severity raised.
  // 7. Return findings + counts.
}
```
