Example coercion helper:
```javascript
function _coerceSubscriptionEntry(entry, manifestEntries) {
  if (!entry) return null;
  // Unwrap legacy double-wrapped entries
  if (entry.name && typeof entry.name === "object" && entry.name.name) {
    return { name: String(entry.name.name), version: String(entry.name.version || entry.version || "0.0.0") };
  }
  // Already flat
  if (typeof entry.name === "string") {
    return { name: entry.name, version: String(entry.version || (manifestEntries.find((m) => m.name === entry.name) || {}).version || "0.0.0") };
  }
  return null;
}
```

```javascript
// ============================================================================
// scaffoldFoundationalPluginData — for foundational/external plugins that
// require pre-existing data.json defaults at install time. Mirrors v0.1.3
// applyTemplaterHotkeys posture (failure-loud + idempotent skip-if-present
// + atomic write + history entries) but writes only when data.json is ABSENT
// (downstream additive helpers handle merge cases for existing files).
//
// Reads:
//   - manifest.foundational_plugins[] (per-item; typically empty)
//   - manifest.external_plugins[] (per-item; lists community-plugin deps)
//   - workshop manifest's foundational_plugins[] (loaded from disk via
//     workshopPath; ensures workshop-level decls drive scaffolding)
//
// Scaffolds: any plugin id that appears in either list AND has a registry
// entry in FOUNDATIONAL_PLUGIN_DEFAULTS.
//
// Idempotent: re-runs are safe (skips when data.json already exists).
// ============================================================================
async function scaffoldFoundationalPluginData(tp, manifest, workshopPath, variables, history, git) {
  // Path-traversal validator on plugin id (mirror v0.21.1 applyCommunityPluginData posture)
  function _validId(id) {
    return typeof id === "string" && /^[a-z0-9][a-z0-9-]*$/i.test(id);
  }

  // Gather candidate plugin ids from per-item manifest
  const perItemIds = new Set();
  for (const dep of manifest.external_plugins || []) {
    if (dep && _validId(dep.id)) perItemIds.add(dep.id);
  }
  for (const dep of manifest.foundational_plugins || []) {
    if (dep && _validId(dep.id)) perItemIds.add(dep.id);
  }

  // Gather workshop-level foundational_plugins[] from disk (read-only)
  const workshopManifestPath = require("path").join(workshopPath, "platform", "manifest.json");
  let workshopFoundational = [];
  try {
    const wsManifest = JSON.parse(require("fs").readFileSync(workshopManifestPath, "utf8"));
    if (Array.isArray(wsManifest.foundational_plugins)) {
      workshopFoundational = wsManifest.foundational_plugins.filter((d) => d && _validId(d.id)).map((d) => d.id);
    }
  } catch (_e) {
    // best-effort; failure to read workshop manifest doesn't block per-item scaffolding
  }

  const candidates = new Set([...perItemIds, ...workshopFoundational]);
  if (candidates.size === 0) return;

  const fs = require("fs");
  const path = require("path");

  for (const id of candidates) {
    if (!FOUNDATIONAL_PLUGIN_DEFAULTS[id]) continue;  // unknown id — silent no-op (no history entry)

    const pluginDir = path.join(tp.app.vault.adapter.getBasePath(), ".obsidian", "plugins", id);
    let pluginDirExists = false;
    try {
      const stat = fs.statSync(pluginDir);
      pluginDirExists = stat.isDirectory();
    } catch (_e) {
      pluginDirExists = false;
    }
    if (!pluginDirExists) {
      history.push({
        time: new Date().toISOString(),
        item: manifest.name,
        step: "scaffold_foundational",
        action: "skipped_missing_plugin_dir",
        id,
        ...(git || {})
      });
      continue;
    }

    const dataPath = path.join(pluginDir, "data.json");
    let dataExists = false;
    try {
      fs.statSync(dataPath);
      dataExists = true;
    } catch (_e) {
      dataExists = false;
    }
    if (dataExists) {
      history.push({
        time: new Date().toISOString(),
        item: manifest.name,
        step: "scaffold_foundational",
        action: "skipped_already_present",
        id,
        ...(git || {})
      });
      continue;
    }

    // Write defaults atomically (tmp + rename). No .sauce-backup since no prior file.
    const defaults = FOUNDATIONAL_PLUGIN_DEFAULTS[id](variables || {});
    const body = JSON.stringify(defaults, null, 2);
    const tmpPath = `${dataPath}.tmp`;
    try {
      fs.writeFileSync(tmpPath, body, "utf8");
      fs.renameSync(tmpPath, dataPath);
      history.push({
        time: new Date().toISOString(),
        item: manifest.name,
        step: "scaffold_foundational",
        action: "scaffolded",
        id,
        path: path.relative(tp.app.vault.adapter.getBasePath(), dataPath),
        ...(git || {})
      });
    } catch (e) {
      new Notice(`scaffoldFoundationalPluginData: write failed (${e.message}) for ${id}`, 8000);
      history.push({
        time: new Date().toISOString(),
        item: manifest.name,
        step: "scaffold_foundational",
        action: "write_failed",
        id,
        error: e.message,
        ...(git || {})
      });
      // Failure-loud: do NOT throw; per-item install continues.
    }
  }
}
```
