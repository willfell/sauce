### Task 2.6: Implement wizard `_autoAddConvenienceIfDvBlueprintsSelected` + wiring

**Files:**
- Modify: `platform/bootstrap-lib/wizard.js`

**Step 1: Locate existing `_buildSubscriptionEntries` + DEFAULT_MECHANISMS_CHECKED for context**

The auto-add helper needs to know about the per-blueprint `depends_on[]`. The wizard already loads workshop manifest (`workshopManifest.blueprints[]`) — but the workshop manifest's blueprint entries are `{name, version, path}` only; the per-blueprint `depends_on[]` lives at `<workshop>/platform/blueprints/<name>/manifest.json`.

Add a helper to load full blueprint manifests:

```javascript
function _loadFullBlueprintManifests(workshopPath, workshopManifest) {
  const fs = require("fs");
  const path = require("path");
  return (workshopManifest.blueprints || []).map(b => {
    try {
      const p = path.join(workshopPath, "platform/blueprints", b.name, "manifest.json");
      return JSON.parse(fs.readFileSync(p, "utf8"));
    } catch (_e) { return null; }
  }).filter(Boolean);
}
```

**Step 2: Add `_autoAddConvenienceIfDvBlueprintsSelected`**

```javascript
function _autoAddConvenienceIfDvBlueprintsSelected(selectedMechs, selectedBlueprints, fullBlueprints) {
  if (selectedMechs.includes("convenience")) return selectedMechs;
  const dvBlueprints = (selectedBlueprints || []).filter(name => {
    const full = (fullBlueprints || []).find(b => b.name === name);
    if (!full || !Array.isArray(full.depends_on)) return false;
    return full.depends_on.some(d => d.name === "convenience");
  });
  if (dvBlueprints.length === 0) return selectedMechs;
  console.log(`[info] Auto-added convenience because ${dvBlueprints[0]} depends on it.`);
  return [...selectedMechs, "convenience"];
}
```

Export both helpers via the `module.exports = {...}` block.

selectedMechs = _autoAddConvenienceIfDvBlueprintsSelected(selectedMechs, selectedBlueprints, fullBlueprints);
